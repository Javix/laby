def run
  loop do
    @players.each do |player|
      system "clear" unless system "cls"
      @game_set.to_s(@players)
      greet_player(player)
      player.query_next_direction(@game_set)
      if player_wins?(player)
        puts "Well done! You did it !"
        exit
      end
    end
  end
end
